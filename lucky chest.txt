-- Hàm tìm rương thông minh với AI
local function findChestsAdvanced()
    local chests = {}
    local radius = FarmSettings.ChestRadius.Value
    local playerPos = RootPart and RootPart.Position or Vector3.new()
    local currentTime = tick()
    
    -- Cache hệ thống để tối ưu
    if cache.lastUpdate + cache.updateInterval > currentTime then
        return cache.chests
    end
    
    -- Quét toàn bộ workspace
    for _, v in pairs(Workspace:GetDescendants()) do
        -- Kiểm tra nếu là rương
        if v:IsA("Tool") and v:FindFirstChild("Handle") then
            local handle = v:FindFirstChild("Handle")
            if handle and handle:FindFirstChild("Chest") then
                local distance = (handle.Position - playerPos).Magnitude
                
                -- Kiểm tra trong bán kính
                if distance <= radius then
                    -- Phân loại rương dựa trên màu sắc hoặc tên
                    local chestType = "Thường"
                    local chestValue = 1000
                    
                    if handle.BrickColor == BrickColor.new("Bright yellow") then
                        chestType = "Vàng"
                        chestValue = 10000
                    elseif handle.BrickColor == BrickColor.new("Bright red") then
                        chestType = "Bạc"
                        chestValue = 5000
                    elseif handle.BrickColor == BrickColor.new("Cyan") then
                        chestType = "Ngọc"
                        chestValue = 50000
                    elseif handle.BrickColor == BrickColor.new("Bright purple") then
                        chestType = "Huyền thoại"
                        chestValue = 100000
                    end
                    
                    -- Lọc theo loại rương
                    local shouldAdd = false
                    local priority = FarmSettings.ChestPriority.Value
                    
                    if priority == "Tất cả" then
                        shouldAdd = true
                    elseif priority == "Rương Thường" and chestType == "Thường" then
                        shouldAdd = true
                    elseif priority == "Rương Bạc" and chestType == "Bạc" then
                        shouldAdd = true
                    elseif priority == "Rương Vàng" and chestType == "Vàng" then
                        shouldAdd = true
                    elseif priority == "Rương Ngọc" and chestType == "Ngọc" then
                        shouldAdd = true
                    elseif priority == "Rương Huyền Thoại" and chestType == "Huyền thoại" then
                        shouldAdd = true
                    end
                    
                    -- Kiểm tra blacklist
                    if shouldAdd and not blacklistedChests[v.Name] then
                        table.insert(chests, {
                            object = v,
                            position = handle.Position,
                            distance = distance,
                            type = chestType,
                            value = chestValue,
                            timeDetected = currentTime
                        })
                    end
                end
            end
        end
    end
    
    -- Sắp xếp theo độ ưu tiên
    local farmMode = PrioritySettings.FarmMode.Value
    
    if farmMode == "Gần nhất" then
        table.sort(chests, function(a, b)
            return a.distance < b.distance
        end)
    elseif farmMode == "Xa nhất" then
        table.sort(chests, function(a, b)
            return a.distance > b.distance
        end)
    elseif farmMode == "Giá trị nhất" then
        table.sort(chests, function(a, b)
            return a.value > b.value
        end)
    elseif farmMode == "Thông minh (AI)" then
        -- AI sorting: ưu tiên rương gần và có giá trị cao
        table.sort(chests, function(a, b)
            local scoreA = a.value / (a.distance + 1)
            local scoreB = b.value / (b.distance + 1)
            return scoreA > scoreB
        end)
    end
    
    -- Cập nhật cache
    cache.chests = chests
    cache.lastUpdate = currentTime
    
    return chests
end

-- Hàm tìm trái ác quỷ
local function findFruits()
    local fruits = {}
    local radius = FarmSettings.FruitRadius.Value
    local playerPos = RootPart and RootPart.Position or Vector3.new()
    local minValue = getFruitValueThreshold(FruitSettings.Value)
    
    for _, v in pairs(Workspace:GetDescendants()) do
        if v:IsA("Tool") and v:FindFirstChild("Handle") then
            local handle = v:FindFirstChild("Handle")
            if handle and handle:FindFirstChild("Fruit") then
                local distance = (handle.Position - playerPos).Magnitude
                
                if distance <= radius then
                    -- Xác định giá trị trái
                    local fruitValue = getFruitValue(v.Name)
                    
                    if fruitValue >= minValue then
                        table.insert(fruits, {
                            object = v,
                            position = handle.Position,
                            distance = distance,
                            value = fruitValue,
                            name = v.Name
                        })
                    end
                end
            end
        end
    end
    
    table.sort(fruits, function(a, b)
        return a.value > b.value
    end)
    
    return fruits
end

-- Hàm tìm boss
local function findBosses()
    local bosses = {}
    local radius = FarmSettings.BossRadius.Value
    local playerPos = RootPart and RootPart.Position or Vector3.new()
    
    local bossNames = {
        "NPC", "Boss", "Enemy", "Starter", "Advanced", "Elite", "Legendary"
    }
    
    for _, v in pairs(Workspace:GetDescendants()) do
        if v:IsA("Model") and v:FindFirstChild("Humanoid") then
            local humanoid = v:FindFirstChild("Humanoid")
            if humanoid and humanoid.Health > 0 then
                local isBoss = false
                
                for _, name in pairs(bossNames) do
                    if v.Name:find(name) then
                        isBoss = true
                        break
                    end
                end
                
                if isBoss then
                    local root = v:FindFirstChild("HumanoidRootPart")
                    if root then
                        local distance = (root.Position - playerPos).Magnitude
                        
                        if distance <= radius then
                            table.insert(bosses, {
                                object = v,
                                position = root.Position,
                                distance = distance,
                                health = humanoid.Health,
                                maxHealth = humanoid.MaxHealth
                            })
                        end
                    end
                end
            end
        end
    end
    
    table.sort(bosses, function(a, b)
        return a.distance < b.distance
    end)
    
    return bosses
end

-- Hàm bay Tween mượt mà
local function tweenToPositionAdvanced(targetPosition, speed)
    if not RootPart then return nil end
    
    local distance = (RootPart.Position - targetPosition).Magnitude
    local time = distance / speed
    
    -- Giới hạn thời gian tối thiểu
    time = math.max(time, 0.1)
    
    local tweenInfo = TweenInfo.new(
        time,
        Enum.EasingStyle.Linear,
        Enum.EasingDirection.Out,
        0,
        false,
        0
    )
    
    local tween = TweenService:Create(
        RootPart,
        tweenInfo,
        {CFrame = CFrame.new(targetPosition)}
    )
    
    tween:Play()
    
    -- Theo dõi quá trình bay
    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not tween or tween.PlaybackState == Enum.PlaybackState.Completed then
            connection:Disconnect()
            return
        end
        
        -- Kiểm tra kẹt
        if config.antiStuck then
            local currentPos = RootPart.Position
            if (currentPos - lastPosition).Magnitude < 1 then
                stuckCounter = stuckCounter + 1
                if stuckCounter > 50 then
                    -- Bị kẹt, thử teleport trực tiếp
                    RootPart.CFrame = CFrame.new(targetPosition)
                    connection:Disconnect()
                    tween:Cancel()
                end
            else
                stuckCounter = 0
            end
            lastPosition = currentPos
        end
    end)
    
    return tween
end

-- Hàm farm chính nâng cao
local function advancedFarming()
    while isFarming and not isPaused do
        -- Kiểm tra an toàn
        if config.safeMode and checkDanger() then
            moveToSafeSpot()
            task.wait(2)
            continue
        end
        
        -- Chọn mục tiêu dựa trên chế độ farm
        local targets = {}
        
        if Options.FarmChest.Value then
            local chests = findChestsAdvanced()
            for _, chest in ipairs(chests) do
                table.insert(targets, {
                    type = "chest",
                    data = chest,
                    priority = chest.value / chest.distance
                })
            end
        end
        
        if Options.FarmFruit.Value then
            local fruits = findFruits()
            for _, fruit in ipairs(fruits) do
                table.insert(targets, {
                    type = "fruit",
                    data = fruit,
                    priority = fruit.value / fruit.distance
                })
            end
        end
        
        if Options.FarmBoss.Value then
            local bosses = findBosses()
            for _, boss in ipairs(bosses) do
                table.insert(targets, {
                    type = "boss",
                    data = boss,
                    priority = 1000000 / boss.distance -- Boss luôn được ưu tiên cao
                })
            end
        end
        
        -- Sắp xếp theo độ ưu tiên
        table.sort(targets, function(a, b)
            return a.priority > b.priority
        end)
        
        -- Farm mục tiêu
        if #targets > 0 then
            for _, target in ipairs(targets) do
                if not isFarming or isPaused then break end
                
                currentTarget = target.data.object
                
                -- Di chuyển đến mục tiêu
                local tween = tweenToPositionAdvanced(
                    target.data.position,
                    FarmSettings.TeleportSpeed.Value
                )
                
                -- Đợi đến nơi
                if tween then
                    tween.Completed:Wait()
                end
                
                -- Tương tác với mục tiêu
                if target.type == "chest" then
                    collectChest(target.data)
                elseif target.type == "fruit" then
                    collectFruit(target.data)
                elseif target.type == "boss" then
                    fightBoss(target.data)
                end
                
                -- Delay giữa các lần farm
                task.wait(FarmSettings.FarmDelay.Value)
            end
        else
            -- Không có mục tiêu, chuẩn bị đổi server
            noChestCount = noChestCount + 1
            
            if noChestCount >= 5 and Options.AutoServerHop.Value then
                logActivity("Không tìm thấy mục tiêu, chuẩn bị đổi server...")
                prepareServerHop()
                noChestCount = 0
            end
        end
        
        task.wait(0.5)
    end
end

-- Hàm nhặt rương
local function collectChest(chestData)
    if not chestData or not chestData.object then return end
    
    local chest = chestData.object
    local handle = chest:FindFirstChild("Handle")
    
    if handle then
        -- Mô phỏng chạm để nhặt rương
        firetouchinterest(handle, RootPart, 0)
        task.wait(0.1)
        firetouchinterest(handle, RootPart, 1)
        
        -- Cập nhật thống kê
        totalFarmedChests = totalFarmedChests + 1
        table.insert(farmedChests, chest)
        
        -- Log hoạt động
        logActivity(string.format("Đã nhặt rương %s tại vị trí %s", 
            chestData.type or "Không xác định",
            tostring(chestData.position)
        ))
        
        -- Thông báo Discord
        if DiscordSettings.EnableWebhook.Value and NotificationSettings.NotifyChest.Value then
            sendDiscordNotification({
                title = "Đã nhặt rương!",
                description = string.format("Loại: %s\nGiá trị: %s\nVị trí: %s",
                    chestData.type or "Không xác định",
                    formatNumber(chestData.value or 1000),
                    tostring(chestData.position)
                ),
                color = 0x00FF00
            })
        end
        
        -- Cập nhật ESP
        if ESPSettings.EnableESP.Value then
            updateESP()
        end
        
        -- Cập nhật UI
        updateStats()
    end
end

-- Hàm đánh boss
local function fightBoss(bossData)
    if not bossData or not bossData.object then return end
    
    local boss = bossData.object
    local bossHumanoid = boss:FindFirstChild("Humanoid")
    local bossRoot = boss:FindFirstChild("HumanoidRootPart")
    
    if bossHumanoid and bossHumanoid.Health > 0 and bossRoot then
        -- Di chuyển đến boss
        tweenToPositionAdvanced(bossRoot.Position + Vector3.new(0, 5, 0), 200)
        
        -- Trang bị vũ khí
        equipWeapon()
        
        -- Tấn công boss
        local attackStart = os.time()
        
        while bossHumanoid and bossHumanoid.Health > 0 and isFarming do
            -- Di chuyển xung quanh boss để tránh đòn
            local randomOffset = Vector3.new(
                math.random(-10, 10),
                0,
                math.random(-10, 10)
            )
            
            RootPart.CFrame = CFrame.new(bossRoot.Position + randomOffset)
            
            -- Tấn công
            virtualClick()
            
            task.wait(0.5)
            
            -- Kiểm tra thời gian đánh boss quá lâu
            if os.time() - attackStart > 60 then
                logActivity("Đánh boss quá lâu, bỏ qua...")
                break
            end
        end
        
        if bossHumanoid and bossHumanoid.Health <= 0 then
            totalFarmedBosses = totalFarmedBosses + 1
            
            logActivity("Đã tiêu diệt boss thành công!")
            
            if DiscordSettings.EnableWebhook.Value and NotificationSettings.NotifyBoss.Value then
                sendDiscordNotification({
                    title = "Đã tiêu diệt boss!",
                    description = string.format("Boss: %s\nVị trí: %s",
                        boss.Name,
                        tostring(bossRoot.Position)
                    ),
                    color = 0xFF0000
                })
            end
        end
    end
end

-- Hàm kiểm tra nguy hiểm
local function checkDanger()
    local playerPos = RootPart.Position
    
    -- Kiểm tra nước
    if playerPos.Y < 0 then
        return true
    end
    
    -- Kiểmểm tra người chơi thù địch
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local character = player.Character
            local root = character:FindFirstChild("HumanoidRootPart")
            local humanoid = character:FindFirstChild("Humanoid")
            
            if root and humanoid and humanoid.Health > 0 then
                local distance = (root.Position - playerPos).Magnitude
                
                -- Nếu có người chơi khác trong bán kính 30 và đang tấn công
                if distance < 30 and isPlayerAttacking(player) then
                    return true
                end
            end
        end
    end
    
    return false
end

-- Hàm kiểm tra người chơi đang tấn công
local function isPlayerAttacking(player)
    -- Kiểm tra animation hoặc trạng thái
    if player.Character then
        local humanoid = player.Character:FindFirstChild("Humanoid")
        if humanoid then
            local animTracks = humanoid:GetPlayingAnimationTracks()
            for _, track in ipairs(animTracks) do
                if track.Animation.AnimationId:find("attack") then
                    return true
                end
            end
        end
    end
    return false
end

-- Hàm trang bị vũ khí
local function equipWeapon()
    local backpack = LocalPlayer.Backpack
    
    -- Tìm vũ khí mạnh nhất
    local bestWeapon = nil
    local bestDamage = 0
    
    for _, item in ipairs(backpack:GetChildren()) do
        if item:IsA("Tool") then
            local damage = getWeaponDamage(item.Name)
            if damage > bestDamage then
                bestDamage = damage
                bestWeapon = item
            end
        end
    end
    
    if bestWeapon then
        LocalPlayer.Character.Humanoid:EquipTool(bestWeapon)
    end
end

-- Hàm format số
local function formatNumber(num)
    if num >= 1000000 then
        return string.format("%.1fM", num / 1000000)
    elseif num >= 1000 then
        return string.format("%.1fK", num / 1000)
    else
        return tostring(num)
    end
end

-- Hàm gửi thông báo Discord
local function sendDiscordNotification(data)

-- Hàm refresh server list nâng cao
local function refreshServerListAdvanced()
    local success, response = pcall(function()
        return HttpService:GetAsync(string.format(
            "https://games.roblox.com/v1/games/%d/servers/Public?limit=100&sortOrder=Asc",
            game.PlaceId
        ))
    end)
    
    if success then
        local data = HttpService:JSONDecode(response)
        serverList = data.data
        
        local serverNames = {}
        for i, server in ipairs(serverList) do
            local ping = math.random(20, 200) -- Simulated ping
            local name = string.format("Server %d | %d/%d | %dms | %s",
                i,
                server.playing,
                server.maxPlayers,
                ping,
                server.id == game.JobId and "Hiện tại" or ""
            )
            table.insert(serverNames, name)
        end
        
        ServerListUI.ServerDropdown:SetValues(serverNames)
        
        -- Cập nhật server hiện tại
        ServerListUI.CurrentServer:SetText(string.format("Server hiện tại: %s", game.JobId))
        ServerListUI.ServerPlayers:SetText(string.format("Người chơi: %d/%d",
            #Players:GetPlayers(),
            serverList[1] and serverList[1].maxPlayers or 0
        ))
        
        logActivity("Đã cập nhật danh sách server")
    else
        logActivity("Không thể lấy danh sách server")
    end
end

-- Hàm chuyển server nâng cao
local function hopToRandomServerAdvanced()
    if #serverList == 0 then
        refreshServerListAdvanced()
        task.wait(2)
    end
    
    local availableServers = {}
    for _, server in ipairs(serverList) do
        if server.id ~= game.JobId and server.playing < server.maxPlayers then
            table.insert(availableServers, server)
        end
    end
    
    if #availableServers > 0 then
        local selectedServer = availableServers[math.random(1, #availableServers)]
        
        logActivity(string.format("Đang chuyển đến server %s...", selectedServer.id))
        
        if DiscordSettings.EnableWebhook.Value and NotificationSettings.NotifyServerHop.Value then
            sendDiscordNotification({
                title = "Đang đổi server",
                description = string.format("Server: %s\nNgười chơi: %d/%d",
                    selectedServer.id,
                    selectedServer.playing,
                    selectedServer.maxPlayers
                ),
                color = 0x00FFFF
            })
        end
        
        totalServerHops = totalServerHops + 1
        TeleportService:TeleportToPlaceInstance(game.PlaceId, selectedServer.id, LocalPlayer)
    else
        logActivity("Không tìm thấy server khả dụng")
    end
end

-- Hàm chuẩn bị đổi server
local function prepareServerHop()
    if not Options.AutoServerHop.Value then return end
    
    -- Kiểm tra giới hạn đổi server
    if config.maxServerHops > 0 and totalServerHops >= config.maxServerHops then
        logActivity("Đã đạt giới hạn đổi server")
        return
    end
    
    -- Đếm ngược
    for i = config.serverHopDelay, 1, -1 do
        if not isFarming then break end
        logActivity(string.format("Đổi server sau %d giây...", i))
        task.wait(1)
    end
    
    if isFarming then
        hopToRandomServerAdvanced()
    end
end

-- Hàm cập nhật ESP
local function updateESP()
    if not ESPSettings.EnableESP.Value then return end
    
    -- Xóa ESP cũ
    for _, v in ipairs(Workspace:GetDescendants()) do
        if v:IsA("BillboardGui") and v.Name == "LuckyESP" then
            v:Destroy()
        end
    end
    
    -- Tạo ESP mới
    spawn(function()
        while ESPSettings.EnableESP.Value do
            -- ESP cho rương
            if ESPSettings.ShowChests.Value then
                for _, chest in ipairs(findChestsAdvanced()) do
                    createESP(chest.object, "📦 " .. chest.type, ESPColors.ChestColor.Value)
                end
            end
            
            -- ESP cho trái
            if ESPSettings.ShowFruits.Value then
                for _, fruit in ipairs(findFruits()) do
                    createESP(fruit.object, "🍎 " .. fruit.name, ESPColors.FruitColor.Value)
                end
            end
            
            -- ESP cho boss
            if ESPSettings.ShowBosses.Value then
                for _, boss in ipairs(findBosses()) do
                    createESP(boss.object, "👾 " .. boss.object.Name, ESPColors.BossColor.Value)
                end
            end
            
            task.wait(1)
        end
    end)
end

-- Hàm tạo ESP
local function createESP(obj, text, color)
    if not obj or not obj:FindFirstChild("Handle") then return end
    
    local handle = obj:FindFirstChild("Handle")
    local billboard = Instance.new("BillboardGui")
    local textLabel = Instance.new("TextLabel")
    
    billboard.Name = "LuckyESP"
    billboard.Adornee = handle
    billboard.Size = UDim2.new(0, 100, 0, 40)
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.AlwaysOnTop = true
    
    textLabel.Parent = billboard
    textLabel.Size = UDim2.new(1, 0, 1, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.Text = text
    textLabel.TextColor3 = color
    textLabel.TextScaled = true
    textLabel.Font = Enum.Font.GothamBold
    textLabel.TextStrokeTransparency = 0.5
    textLabel.TextStrokeColor3 = Color3.new(0, 0, 0)
    
    billboard.Parent = handle
end
